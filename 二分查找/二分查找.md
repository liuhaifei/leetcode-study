public int getIndex(int[] num,int target){
    if(num==null || num.length==0){
        return -1;
    }
    int mid;
    int start=0;
    int end=num.lenth-1;
    while(start+1<end){
        mid=start+（end-start）/2;
        if(num[mid]==target){
            return mid;
        }
        if(target<num[mid]){
            end=mid;
        }else{
            start=mid;
        }
    }
    if(num[start]==target){
        return start;
    }
    if(num[end]==target){
        return end;
    }
    return -1;
}

   public int getNums(int[] nums,int least){
       if(nums==null || nums.length==0){
           return 0;
       }
       int start=0;
       int end=getMax(nums);
       int mid;
       while(start+1<end){
           mid=start+(end-start)/2;
           if(getPiece(nums,mid)>least){
               start=mid;
           }else{
               end=mid;
           }
       }
       if(getPiece(nums,end)>=least){
           return end;
       }
       if(getPiece(nums,start)>=least){
           return start;
       }
       return 0;
   }
